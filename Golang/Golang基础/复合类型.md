# 复合类型

```go
package main

import "fmt"

// 结构体
type Student struct {
	name   string
	gender string
	score  int
}

func main() {
	// 数组是一系列同类型数据集合
	arr := [5]int{4, 3, 2, 5, 1}
	for _, v := range bubbleSort(arr) {
		fmt.Println(v)
	}

	// 切片长度不固定
	slice := []int{1, 2, 3, 4, 5, 6, 7}
	// copy是将后者复制到前者，以长度短的为准
	// 数组[起点:终点:容量]
	copy(slice, arr[:5:5])
	fmt.Println(slice)

	// 字典
	// make只适用于内置数据结构，即map、channel和slice，返回带有非零初值的类型本身。new适用于所有，返回指针
	dict := make(map[string]int, 3)
	dict["one"] = 1
	delete(dict, "one")

	student := Student{
		name:   "Lili",
		gender: "girl",
		score:  0,
	}
	student.learn()
}

// 指针用来存储任何一个变量的内存地址
// 不要操作没有合法执行的内存
func (s *Student) learn() {
	if s.score <= 100 {
		s.score += 10
	}
}

// 冒泡排序，时间复杂度O(n^2)
func bubbleSort(arr [5]int) [5]int {
	for i := 0; i < len(arr)-1; i++ {
		for j := 0; j < len(arr)-1-i; j++ {
			if arr[j] > arr[j+1] {
				arr[j], arr[j+1] = arr[j+1], arr[j]
			}
		}
	}
	return arr
}

// 切片删除
func sliceDelete(slice []int, startIndex, endIndex int) []int {
	// append是将后者往前者增加
	return append(slice[:startIndex], slice[endIndex:]...)
}

// 切片插入
func sliceInsert(slice []int, index int, args ...int) []int {
	return append(append(slice[:index], args...), slice[index:]...)
}
```
