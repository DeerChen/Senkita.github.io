# 流程控制

避免多层嵌套，建议控制在 3 层以内，否则影响性能。

```go
package main

import (
	"fmt"
)

var i int

func main() {
	fmt.Scanf("%d", &i)

	// if结构
	if i > 0 {
		fmt.Printf("%d > 0", i)
	} else if i < 0 {
		fmt.Printf("%d < 0", i)
	} else {
		fmt.Printf("%d = 0", i)
	}
	// switch结构
	switch true {
	// case自带break
	case i > 0:
		fmt.Printf("%d > 0", i)
	case i < 0:
		fmt.Printf("%d < 0", i)
	case i == 0:
		fmt.Printf("%d = 0", i)
		// 继续执行
		fallthrough
	default:
		fmt.Println(i)
	}

	arr := []string{
		"10",
		"20",
		"30",
	}
	// for range循环
	for _, v := range arr {
		if v == "10" {
			// 跳出本次循环，继续下次循环
			continue
		} else {
			// 跳出循环
			break
		}
	}
	// for循环
	for i := 0; i < 10; i++ {
		// 跳转标签
		goto LastWord
	}
LastWord:
	fmt.Println("End")
}
```
